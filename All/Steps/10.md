# Step 10 — Cost Calculation

**Phase:** 4 — Cost Engine  
**Input:** Machining time (Step 8) + Prediction interval (Step 9) + Stock weight (Step 6) + User inputs (Step 3) + Final routing + machine selections (Step 4)  
**Output:** Full cost breakdown (material + machining + setup + overhead) + machining breakdown by process

---

## Goal

Calculate all four cost components using deterministic formulas. Only machining cost depends on the ML prediction — everything else is formula/rule-based.

Key architecture note (judge-proof):

> The ML model predicts total machining time, while the cost engine allocates that time across active processes using heuristic weights to enable machine-specific costing.

## Material Cost

```python
PRICE_PER_KG = {
    "Aluminium": 220,    # ₹/kg
    "Steel":     85,
    "Brass":     450,
    "Titanium":  3200,
    "ABS":       150,
}

BATCH_DISCOUNT = {
    (1, 9):    1.00,
    (10, 50):  0.90,
    (51, 200): 0.80,
    (201, float('inf')): 0.70,
}

def get_batch_discount(quantity: int) -> float:
    if quantity <= 0:
        raise ValueError("quantity must be >= 1")
    for (low, high), disc in BATCH_DISCOUNT.items():
        if low <= quantity <= high:
            return disc
    # Should never happen
    return 1.0

def material_cost(stock_weight_kg, material, quantity):
    price = PRICE_PER_KG[material]
    discount = get_batch_discount(quantity)
    return round(stock_weight_kg * price * discount, 2)
```

## Machining Cost

```python
RATE_PER_MACHINE_PER_HOUR = {
    # Example shop rates (₹/hour). These are configurable.
    "3-Axis VMC": 1000,
    "4-Axis":     1300,
    "5-Axis":     1800,
    "CNC Lathe":  1200,
    "Manual Lathe": 700,
    "Surface Grinder": 900,
    "VMC Drill":  900,
}

# Simple time allocation across active processes.
# This keeps the model output as a single total time while still enabling
# machine-specific costing when the engineer changes a machine.
# NOTE: these weights are a heuristic representing typical time distribution
# across processes for general mechanical parts; they are not learned from data.
PROCESS_TIME_WEIGHTS = {
    "turning":  0.35,
    "milling":  0.45,
    "drilling": 0.15,
    "grinding": 0.05,
}

def allocate_time_minutes(machining_time_min: float, routing_final: dict) -> dict:
    active = {
        "turning":  int(routing_final.get("is_turning", 0)),
        "milling":  int(routing_final.get("is_milling", 0)),
        "drilling": int(routing_final.get("is_drilling", 0)),
        "grinding": int(routing_final.get("is_grinding", 0)),
    }
    total_weight = sum(PROCESS_TIME_WEIGHTS[p] for p, on in active.items() if on)
    if total_weight <= 0:
        return {p: 0.0 for p in active}
    return {
        p: round(machining_time_min * (PROCESS_TIME_WEIGHTS[p] / total_weight), 2) if on else 0.0
        for p, on in active.items()
    }

def machining_cost(machining_time_min: float, routing_final: dict, machine_selection_final: dict) -> float:
    time_split = allocate_time_minutes(machining_time_min, routing_final)

    total = 0.0
    for process, minutes in time_split.items():
        if minutes <= 0:
            continue
        machine = machine_selection_final.get(process)
        if not machine:
            raise ValueError(f"Missing machine selection for active process: {process}")
        rate_per_hour = RATE_PER_MACHINE_PER_HOUR[machine]
        total += minutes * (rate_per_hour / 60.0)

    return round(total, 2)


def machining_cost_breakdown(machining_time_min: float, routing_final: dict, machine_selection_final: dict) -> dict:
    """Return per-process machining minutes and ₹ contribution for transparency."""
    time_split = allocate_time_minutes(machining_time_min, routing_final)
    breakdown = {}
    for process, minutes in time_split.items():
        if minutes <= 0:
            continue
        machine = machine_selection_final.get(process)
        if not machine:
            raise ValueError(f"Missing machine selection for active process: {process}")
        rate_per_hour = RATE_PER_MACHINE_PER_HOUR[machine]
        cost = round(minutes * (rate_per_hour / 60.0), 2)
        breakdown[process] = {
            "machine": machine,
            "minutes": round(minutes, 2),
            "rate_per_hour": rate_per_hour,
            "cost": cost,
        }
    breakdown["total_machining_cost"] = round(sum(v["cost"] for k, v in breakdown.items() if k != "total_machining_cost"), 2)
    return breakdown
```

## Setup Cost

```python
import math

def setup_cost(quantity):
    return round(max(2000 / math.sqrt(quantity), 200), 2)
```

Large batches distribute setup cost. Floor of ₹200 prevents zero setup.

Intuition: setup effort decreases **sub-linearly** with batch size because fixture preparation, calibration, and tool setup happen once per batch, then get amortized.

## Overhead

```python
OVERHEAD_RATE = {
    "Aluminium": 0.12,
    "Steel":     0.15,
    "Brass":     0.18,
    "Titanium":  0.22,
    "ABS":       0.10,
}

def overhead_cost(material_cost, machining_cost, setup_cost, material):
    subtotal = material_cost + machining_cost + setup_cost
    rate = OVERHEAD_RATE[material]
    return round(subtotal * rate, 2)
```

Rationale (why overhead varies by material): harder-to-machine or higher-spec materials typically require more tooling wear allowance, inspection, QA handling, and process controls, which increases overhead.

## Total Quote

```python
def calculate_total_quote(stock_weight_kg, machining_time_min,
                          material, quantity,
                          routing_final, machine_selection_final):
    if quantity <= 0:
        raise ValueError("quantity must be >= 1")
    if stock_weight_kg <= 0:
        raise ValueError("stock_weight_kg must be > 0")
    if material not in PRICE_PER_KG:
        raise ValueError("Unsupported material")

    mat  = material_cost(stock_weight_kg, material, quantity)
    mach = machining_cost(machining_time_min, routing_final, machine_selection_final)
    stp  = setup_cost(quantity)
    ovhd = overhead_cost(mat, mach, stp, material)
    
    total = mat + mach + stp + ovhd
    
    return {
        "material_cost":  mat,
        "machining_cost": mach,
        "setup_cost":     stp,
        "overhead_cost":  ovhd,
        "total_cost":     round(total, 2),
    }

def calculate_cost_range(stock_weight_kg, interval, material, quantity,
                         routing_final, machine_selection_final):
    """Convert machining time interval to a total cost interval."""
    # Deterministic components computed once
    mat  = material_cost(stock_weight_kg, material, quantity)
    stp  = setup_cost(quantity)

    mach_low  = machining_cost(interval["lower_time_min"], routing_final, machine_selection_final)
    mach_high = machining_cost(interval["upper_time_min"], routing_final, machine_selection_final)

    ovhd_low  = overhead_cost(mat, mach_low,  stp, material)
    ovhd_high = overhead_cost(mat, mach_high, stp, material)

    return {
        "total_cost_lower": round(mat + mach_low  + stp + ovhd_low,  2),
        "total_cost_upper": round(mat + mach_high + stp + ovhd_high, 2),
    }
```
